    print(f"   ---> {len(reviews)} total reviews")

def text_preprocessing(text):
    
    # Converting words to lower case
    text = text.lower()
        
    # Formatting words and removing unwanted characters
    text = re.sub(r'https?:\/\/.*[\r\n]*', '', text, flags=re.MULTILINE)
    text = re.sub(r'\<a href', ' ', text)
    text = re.sub(r'&amp;', '', text) 
    text = re.sub(r'[_"\-;%()|+&=*%.,!?:#$@\[\]/]', ' ', text)
    text = re.sub(r'<br />', ' ', text)
    text = re.sub(r'\'', ' ', text) 

    # Tokenizing each word
    text = nltk.WordPunctTokenizer().tokenize(text)

    # Lemmatizing each word
    text = [nltk.stem.WordNetLemmatizer().lemmatize(token, pos='v') for token in text if len(token)>1]
    return text

    # Removing stopwords
    stop_words = set(stopwords.words('english'))
    filtered_txt = [t for t in text if t.lower() not in stop_words]
    filtered_text = ' '.join(filtered_txt)
    text=filtered_text

Step 4: Topic Modeling using the Latent Dirichlet Allocation (LDA)¶

import gensim
from gensim import corpora, models

# Select the column you want to perform topic modeling on
texts = df['processed_reviews']

# Create a dictionary from the tokens
dictionary = corpora.Dictionary(texts)

# Convert the tokens to a bag-of-words corpus
corpus = [dictionary.doc2bow(text) for text in texts]

# Train an LDA model with 10 topics
lda_model = models.ldamodel.LdaModel(corpus=corpus, num_topics=10, id2word=dictionary, passes=10)

topic_names = []
for i in range(10):
    top_words = [w for w, _ in lda_model.show_topic(i, topn=5)]
    topic_names.append('Topic {}: {}'.format(i, ', '.join(top_words)))

# Print the topics and their associated words
for i, topic in enumerate(lda_model.print_topics(-1)):
    print(topic_names[i])
    print('Words:', topic[1], '\n')

from textblob import TextBlob

# Create a function to apply TextBlob sentiment analysis to each row of the dataframe
def sentiment_analysis(text):
    blob = TextBlob(text)
    sentiment_score = blob.sentiment.polarity
    if sentiment_score > 0:
        return 'Positive Feedback'
    elif sentiment_score < 0:
        return 'Negative Feedback'
    else:
        return 'Neutral Feedback'

# Apply the sentiment_analysis function to each row of the dataframe
df['sentiment'] = df['reviews'].apply(sentiment_analysis)
sentiment_counts = df['sentiment'].value_counts()
print(sentiment_counts)

Interpretation of the Results:¶

After scrapping the data, cleaning, and processing it, we conclude that out of the reviews that are not yet validated, almost 68% are positive feedback about the flight experiences of the customers with British Airways. Only one customer had a neutral review, while the rest faced problems with their flights. The wordcloud shows us that the most recurrent words are, as one would expect, "flight" and "BA" referring to British Airways. We can also notice a high frequency of appearance of the words: "seat", "time","hour", "service", "food" which gives us an idea about possible reasons of the customers' dissatisfaction. As a matter of fact, we can think of flight delays as a major reason behind the clients' negative feedback. In terms of locations mentioned, the most obvious ones are "London" and "Heathrow" which can easily be anticipated since the greatest activity of the company is in Britain.
